üí° Example Usage

Swift/iOS

import Template

// Async echo with error handling
do {
    let result = try await echoAsync(
        input: "Hello",
        token: nil
    )
    print(result?.text ?? "")
} catch let error as TemplateError {
    switch error {
    case .InputTooLarge(let size, let max, _):
        print("Too large: \(size) > \(max)")
    case .InvalidInput(let msg, _):
        print("Invalid: \(msg)")
    case .OperationCancelled(let op):
        print("Cancelled: \(op)")
    }
}

Kotlin/Android

import uniffi.rust_multiplatform_template_lib.*

// Configuration with validation
val config = TemplateConfig(
    maxInputSize = 1000u,
    enableValidation = true
)

try {
    val result = config.validateAndEcho("test")
    result?.let {
        println("${it.text} (${it.length})")
    }
} catch (e: TemplateException) {
    when (e) {
        is TemplateException.InputTooLarge ->
            println("Too large")
        is TemplateException.InvalidInput ->
            println(e.message)
    }
}
